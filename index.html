What are polymorphic associations?


"A relationship from one class to multiple classes." 

Person = DS.Model.extend({
  name: attr('string'),
  neckbeard: true
});
            

Person = DS.Model.extend({
  name: attr('string'),
  neckbeard: true,
  pets: hasMany('pet', { polymorphic: true })
});
            

Person = DS.Model.extend({
  name: attr('string'),
  neckbeard: true,
  pets: hasMany('pet', { polymorphic: true })
});

Pet = DS.Model.extend({
  owner: belongsTo('person')
});

Cat = Pet.extend({
  lives: attr('number', {defaultValue: 9})
});
